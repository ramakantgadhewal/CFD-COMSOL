\section{Rate of strain and rate of rotation}

\begin{equation}
	u_{i,j} = S_{ij} + \Omega_{ij}
\end{equation}
where 
\begin{equation}
	S_{ij} = \frac{1}{2}(u_{i,j} + u_{j,i}  )
\end{equation}
(symmetric)
\begin{equation}
	\Omega_{ij} = \frac{1}{2}(u_{i,j} - u_{j,i}  )
\end{equation}
(antisymmetric)
